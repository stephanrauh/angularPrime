$scope.value = 0;

$scope.timedValue = 0;

$scope.doStep = function() {
    $scope.timedValue = $scope.timedValue + 10;
    $timeout(function() {
        if ($scope.timedValue < 100) {
            $scope.doStep();
        }
    }, 1000);
};

$scope.startTimer = function() {
    $scope.timedValue = 0;
    $scope.doStep();
}
