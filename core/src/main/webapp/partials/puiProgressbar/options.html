    $scope.smoothProgress = {
        value : 0,
        showLabel : false,
        easing: 'linear',
        effectSpeed: 1000
    }

    $scope.doStep = function() {
        $scope.smoothProgress.value = $scope.smoothProgress.value + 10;
        $timeout(function() {
            if ($scope.smoothProgress.value < 100) {
                $scope.doStep();
            }
        }, 1000);
    };

    $scope.startTimer = function() {
        $scope.smoothProgress.value = 0;
        $scope.doStep();
    }
