$scope.remoteData = function (callback) {
    $.ajax({
        type: "GET",
        url: 'json/cars.json',
        dataType: "json",
        context: this,
        success: function (response) {
            $scope.safeApply(  // external changes aren't picked up by angular
                callback.call(this, response)
            )
        }
    });
};
