    $scope.remoteData = {
        nodes: function (ui, callback) {
            $.ajax({
                type: "GET",
                url: 'json/tree.json',
                dataType: "json",
                context: this,
                success: function (response) {
                    $scope.safeApply(  // external changes aren't picked up by angular
                        callback.call(this, response))
                }
            });
        },
        nodeSelect : function(event, ui) {
            alert('Node Selected - Data: ' + ui.data);
        }
    };
