		When this notify is closed, a callback is executed.
	

						
	$scope.notifyCallbackOptions = {
		visible : false
		,callback : function () {
			alert('Notify widget closed by user');
		}
	};

	$scope.showWithCallback = function() {
		$scope.notifyCallbackOptions.visible = true;
	};
